# cloud-music-js

## src目录结构如下

└── src  
    ├── App.vue      # 根组件  
    ├── app.js        # 根组件的渲染数据  
    ├── main.js      # Vue文件入口  
    ├── api         # 请求后端数据的接口  
    ├── assets          # 静态资源文件  
    ├── components         # 子组件  
    │   ├── billboard  
    │   ├── header  
    │   ├── lyric  
    │   ├── musicList  
    │   └── musicPlayer  
    ├── filter           # 格式化公用函数  
    ├── i18n            # 不同国家语言  
    ├── model          # 实体类  
    ├── router          # 路由  
    ├── store          # 组件之间共享的数据存储仓库  
    └── util          # 工具函数
